CREATE TABLE users(
             user_id CHAR(36) DEFAULT (UUID()),
             user_name VARCHAR(30) NOT NULL,
             email VARCHAR(30) NOT NULL,
             password VARCHAR(150) NOT NULL,
             grade INT NOT NULL,
             contact_number CHAR(10),
             address VARCHAR(80),
             CONSTRAINT pk_users PRIMARY KEY (user_id));

CREATE TABLE available_books(
             book_id CHAR(36) DEFAULT (UUID()),
             book_name VARCHAR(50) NOT NULL,
             author VARCHAR(30),
             version CHAR(7),
             filename VARCHAR(100),
             uploader_id VARCHAR(36),
             requester_id VARCHAR(36),
             status VARCHAR(10) DEFAULT 'Available',
             CONSTRAINT pk_avbooks PRIMARY KEY (book_id),
             CONSTRAINT fk_avbooks_1 FOREIGN KEY (uploader_id) REFERENCES users(user_id),
             CONSTRAINT fk_avbooks_2 FOREIGN KEY (requester_id) REFERENCES users(user_id));
